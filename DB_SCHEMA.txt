TABLE: accounts

+----------------------------------------------------+
| id (PRIMARY KEY) | balance | salary_date | role_id |
+----------------------------------------------------+
|      big int     | big int |     int     | big int |
+----------------------------------------------------+

CREATE TABLE accounts (
    id BIGINT PRIMARY KEY,
    balance BIGINT,
    salary_date INTEGER,
    role_id BIGINT
);


TABLE: roles

+-----------------------------+
| id (PRIMARY KEY) |  salary  |
+-----------------------------+
|      big int     | big int  |
+-----------------------------+

CREATE TABLE roles (
    id BIGINT PRIMARY KEY,
    salary BIGINT
);


TABLE: bets

+------------------+----------+---------+--------+
| id (PRIMARY KEY) |  result  |   pot   |  open  |
+------------------+----------+---------+--------+
|      big int     |    int   | big int |  bool  |
+------------------+----------+---------+--------+

CREATE TABLE bets (
    id integer primary key autoincrement, 
    result int, 
    pot bigint, 
    open BOOLEAN
);

TABLE: gamblers

+------------------+--------------+------------+----------+----------+
| id (PRIMARY KEY) |  account_id  |   bet_id   |  choice  |  amount  |
+------------------+--------------+------------+----------+----------+
|      big int     |    big int   |   big int  |    int   |  big int |
+------------------+--------------+------------+----------+----------+

CREATE TABLE gamblers (
    id integer primary key autoincrement, 
    account_id bigint, 
    bet_id bigint, 
    choice int,
    amount bigint
);
